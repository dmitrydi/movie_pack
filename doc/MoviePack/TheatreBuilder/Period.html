def initialize(interv, &block)
  @interv = interv
  @description = MoviePack::Theatre::DEFAULT_DESCRIPTION
  @filters = {}
  @price = MoviePack::Theatre::DEFAULT_PRICE
  @hall = []
  instance_eval(&block) if block_given?
end
